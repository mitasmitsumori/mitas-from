// 例：構造に応じた坪単価設定（1坪 = 3.3㎡で換算） const unitPrices = { "木造": 33000 / 3.3, "RC造": 50000 / 3.3, "S造": 33000 / 3.3, "軽量鉄骨造": 33000 / 3.3 }; const area = parseFloat(params.get("area") || 0); const structure = params.get("structure") || "木造"; const demolitionCost = Math.round(area * unitPrices[structure]); // 表に追加表示 const table = document.querySelector("table"); const row = table.insertRow(-1); row.innerHTML = `解体工事費${demolitionCost.toLocaleString()} 円`;
